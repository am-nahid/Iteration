Check whether a Number is a prime or not.

const Prime_Check = (N) => 
{
	let factor=0;
	for(i=1;i<=N;i++){
	  if(N%i===0){
	   factor++;
	  }
	  }
	  if (factor===2){
	    return ("YES");
	  }
	  else{
	  return ("NO");
	}
	
	
	
	
};
